# Indo-Compiler For Click.cl



HTML:--

```html
   <div>
      <ul c-loop='main>>a'>
         <ul c-loop='a.main>>a'>
             <li>{a.name}</li>
         </ul>
      </ul>
  </div>
```
   
#Output :-

```js
function view(_, cX) {
    return (
        cX.render("div", {}, 
            cX.render("ul", {}, 
                _.main.map(function(a) {
                    return (
                        cX.render("ul", {}, 
                                a.main.map(function(a) {
                                     return (cX.render("li", {}, a.name))
                                })
                        )
                    )
                })
            )
        )
    )
}
```
